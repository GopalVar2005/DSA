Problem Statement: Given an integer array arr of size N, sorted in ascending order (with distinct values). Now the array is rotated between 1 to N times which is unknown. Find how many times the array has been rotated. 

Application of Find Min in rotated sorted array

Appr: jitni place se min ko shift kiya ha utni baar usse rotate kiya hoga
import java.util.*;
class Solution {
    public int findKRotation(int arr[]) {
        // Code here
        int lo=0;
        int hi=arr.length-1;
        int min=Integer.MAX_VALUE;
        int minidx=-1;
        while(lo<=hi)
        {
            int mid=lo+(hi-lo)/2;
            if(arr[lo]<=arr[mid])
            {
                if(arr[lo]<min)
                {
                    min=arr[lo];
                    minidx=lo;
                }
                lo=mid+1;
            }
            else
            {
                if(arr[mid]<min)
                {
                    min=arr[mid];
                    minidx=mid;
                }
                hi=mid-1;
            }
        }
        return minidx;
    }
TC-O(logN)
SC-O(1)
}
