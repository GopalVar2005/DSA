Problem Statement:  Given a sorted array, arr[] and a number target, you need to find the number of occurrences of target in arr[]. 

Appr 1: Use for loop

Appr2: Use first and last BS to get range of numbers
class Solution {
    int countFreq(int[] arr, int target) {
        int first=search(arr, target, true);
        if(first==-1)
        {
            return 0;
        }
        return search(arr, target, false)-first+1;
    }
    public int search(int[] arr, int target, boolean first)
    {
        int lo=0;
        int hi=arr.length-1;
        int ans=-1;
        while(lo<=hi)
        {
            int mid=lo+(hi-lo)/2;
            if(arr[mid]==target)
            {
                ans=mid;
                if(first)
                {
                    hi=mid-1;
                }
                else
                {
                    lo=mid+1;
                }
            }
            else if(arr[mid]<target)
            {
                lo=mid+1;
            }
            else
            {
                hi=mid-1;
            }
        }
        return ans;
    }
TC-O(2*logN)
SC-O(1)
}
